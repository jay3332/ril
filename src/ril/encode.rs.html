143/// Manually configured encoder metadata. This is used to provide fine-grained control over the
144/// encoder. Not all encoders will use/consider all of these options.
145///
146/// # See Also
147/// - [`Encoder::prepare`] for more information on how to use this.
148/// - [`Encoder::encode_static`] to encode a static [`Image`].
149/// - [`Encoder::encode_sequence`] to encode an [`ImageSequence`].
150#[derive(Clone, Debug, PartialEq, Eq, Hash)]
151pub struct EncoderMetadata<P: Pixel> {
152    /// The width of the image.
153    pub width: u32,
154    /// The height of the image.
155    pub height: u32,
156    /// Information regarding whether the image is static (not animated) or not.
157    /// * If `Some`, the image is animated and holds `(frame_count, loop_count)`.
158    /// * If `None`, the image is considered static.
159    ///
160    /// # Note
161    /// If you do not know the frame count or loop count, substitute `0` for the frame count and
162    /// [`LoopCount::Infinite`] for the loop count. All encoders should be able to "grow" this
163    /// information as they encode more frames. (This is the [`default`][Default::default] value.)
164    pub sequence: Option<(usize, LoopCount)>,
165    /// The color type of the image.
166    pub color_type: ColorType,
167    /// The bit depth of the pixels in the image.
168    pub bit_depth: u8,
169    /// The palette of the image.
170    pub palette: Option<Box<[P::Color]>>,
171}

354/// Low-level encoder interface around an image format. Typically only accessed for lazy encoding.
355///
356/// # Encoder metadata
357///
358/// When preparing an encoder, you must pass in some metadata to the encoder. This metadata is
359/// used to configure the encoder for encoding the image. The metadata can be:
360///
361/// * A static [`Image`], in which case the encoder will be prepared for encoding a static image.
362/// * A [`Frame`], in which case the encoder will be prepared for encoding an image sequence.
363/// * An [`ImageSequence`], in which case the encoder will be prepared for encoding an image
364///   sequence.
365/// * An [`EncoderMetadata`] object for more fine-grained control over the encoder.
366///
367/// ## Notes
368///
369/// - The specific metadata that is actually used and/or considered depends on the
370///   encoder implementation.
371/// - If you provide an image, frame, or sequence as metadata, you are only registering metadata,
372///   the image or frame itself will **not** be encoded yet. You would have to make a call to
373///   [`Self::encode_frame`] to encode the image or frame.
374/// - Use [`EncoderMetadata::from`] to create encoder metadata from an image, frame, or sequence.
375///
376/// ## Examples
377///
378/// ```no_run
379/// use ril::encodings::png::{PngEncoder, PngEncoderOptions};
380/// # use ril::prelude::*;
381///
382/// # fn main() -> ril::Result<()> {
383/// # let writer = std::io::stdout();
384/// // Create a new encoder with default options given the width, height, and pixel type:
385/// let encoder = PngEncoder::new(writer, EncoderMetadata::<Rgba>::new(256, 256));
386///
387/// // Create a new encoder will encode an animated image with 2 frames, looping infinitely:
388/// let metadata = EncoderMetadata::<Rgba>::new(256, 256).with_sequence(2, LoopCount::Infinite);
389/// # let writer = std::io::stdout();
390/// let encoder = PngEncoder::new(writer, metadata);
391///
392/// // Using an image/frame/sequence as metadata:
393/// # let writer = std::io::stdout();
394/// let image = Image::<Rgb>::open("sample.png")?; // can be any Image, Frame, or ImageSequence
395/// let encoder = PngEncoder::new(writer, &image);
396///
397/// // Using a frame as baseline metadata but explicitly anticipating frame count
398/// // NOTE: See "Anticipating the frame and loop counts" below for more information
399/// let frame = Frame::from_image(image);
400/// # let writer = std::io::stdout();
401/// let metadata = EncoderMetadata::from(&frame).with_sequence(2, LoopCount::Infinite);
402/// let encoder = PngEncoder::new(writer, metadata);
403///
404/// # Ok(())
405/// # }
406/// ```
407///
408/// # Anticipating frame and loop counts
409///
410/// All frame and loops counts for the following encoders **must** be anticipated prior to
411/// encoding:
412///
413/// * [`PngEncoder`]
414///
415/// This is a limitation caused by the way these encoders/codecs are implemented. For example, in
416/// the PNG format, the frame and loop counts are stored in the header of the file, which is
417/// written at the very beginning of encoding.
418///
419/// You can anticipate the frame and loop counts by using [`EncoderMetadata::with_sequence`].
420/// If you have a reference to an image, frame, or sequence, you can use [`EncoderMetadata::from`]
421/// to create encoder metadata from it:
422///
423/// ```no_run
424/// use ril::encodings::png::{PngEncoder, PngEncoderOptions};
425/// # use ril::prelude::*;
426///
427/// # fn main() -> ril::Result<()> {
428/// # let writer = std::io::stdout();
429/// // Grab our image, frame, or image sequence:
430/// let frame = Frame::from_image(Image::<Rgb>::open("sample.png")?);
431/// // Convert it into encoder metadata with `EncoderMetadata::from`,
432/// // then anticipate the frame and loop counts with `EncoderMetadata::with_sequence`:
433/// let metadata = EncoderMetadata::from(&frame).with_sequence(2, LoopCount::Infinite);
434/// // Finally, create the encoder with our modified metadata:
435/// let encoder = PngEncoder::new(writer, metadata);
436/// # Ok(())
437/// # }
438/// ```
439///
440/// If you are not certain how many frames or loops your image will have, you can use an upper bound
441/// for the frame count or loop count.
442///
443/// # Example
444/// Lazily encode images into an animated PNG (APNG):
445///
446/// ```no_run
447/// use std::fs::File;
448/// use std::time::Duration;
449/// use ril::colors::{RED, BLUE};
450/// use ril::encodings::png::PngEncoder;
451/// use ril::prelude::*;
452///
453/// const HALF_SECOND: Duration = Duration::from_millis(500);
454///
455/// fn main() -> ril::Result<()> {
456///     let mut writer = File::create("output.png")?;
457///     // Prepare metadata. The PNG encoder requires we anticipate the frame count prior to encoding:
458///     let metadata = EncoderMetadata::<Rgb>::new(100, 100).with_frame_count(2);
459///     // Create the encoder
460///     let mut encoder = PngEncoder::new(&mut writer, metadata)?;
461///
462///     // Frame 1: 100x100 red square for 0.5s
463///     let image = Image::new(100, 100, RED);
464///     encoder.add_frame(&Frame::from_image(image).with_delay(HALF_SECOND))?;
465///
466///     // Frame 2: 100x100 blue square for 0.5s
467///     let image = Image::new(100, 100, BLUE);
468///     encoder.add_frame(&Frame::from_image(image).with_delay(HALF_SECOND))?;
469///
470///     // Finish encoding
471///     encoder.finish()
472/// }
473/// ```
474pub trait Encoder<P: Pixel, W: Write>: Sized {
475    /// The type of the configuration object used to configure options for the encoder.
476    type Config: Default;
477
478    /// Creates a new encoder and prepares it for encoding an image.
479    ///
480    /// **This method is only used for lazy encoding of image sequences.**
481    /// - If you are encoding a static image, use [`Self::encode_static`] instead. There is no need to
482    ///   call any intermediate method for encoding static images since they inherently cannot be
483    ///   lazily encoded after all.
484    /// - If you are encoding an [`ImageSequence`] directly, use [`Self::encode_sequence`] instead.
485    ///   Since [`ImageSequence`] already contains all the frames, there is no need to lazily encode
486    ///   them.
487    ///
488    /// # Metadata
489    /// The `metadata` parameter is used to provide initial metadata to the encoder. See the
490    /// [trait-level documentation][Encoder] for more information.
491    ///
492    /// ## Notes
493    /// - The specific metadata that is actually used and/or considered depends on the
494    ///   encoder implementation.
495    /// - If you provide an image or frame as metadata, you are only registering metadata, the
496    ///   image or frame itself will **not** be encoded yet. You would have to make a call to
497    ///   [`Self::encode_frame`] to encode the image or frame.
498    ///
499    /// # Errors
500    /// * An error occured during encoding.
501    /// * The image is invalid for encoding.
502    fn new(dest: W, metadata: impl HasEncoderMetadata<Self::Config, P>) -> crate::Result<Self>;
503
504    /// Adds a frame to the encoding sequence and encodes it into the given writer.
505    ///
506    /// **This method is only used for lazy encoding of image sequences.**
507    /// - If you are encoding a static image, use [`Self::encode_static`] instead. There is no need to
508    ///   call any intermediate method for encoding static images since they inherently cannot be
509    ///   lazily encoded after all.
510    /// - If you are encoding an [`ImageSequence`] directly, use [`Self::encode_sequence`] instead.
511    ///   Since [`ImageSequence`] already contains all the frames, there is no need to lazily encode
512    ///   them.
513    ///
514    /// # Errors
515    /// * An error occured during encoding.
516    fn add_frame(&mut self, frame: &impl FrameLike<P>) -> crate::Result<()>;
517
518    /// Finishes encoding the image. This **must** be called once after all frames are encoded.
519    ///
520    /// # Errors
521    /// * An error occured during encoding or writing.
522    fn finish(self) -> crate::Result<()>;
523
524    /// Encodes a static image into the given writer.
525    ///
526    /// # Errors
527    /// * An error occured during encoding.
528    fn encode_static(image: &Image<P>, dest: W) -> crate::Result<()> {
529        let mut encoder = Self::new(dest, image)?;
530        encoder.add_frame(image)?;
531        encoder.finish()
532    }
533
534    /// Encodes the given image sequence into the given writer.
535    ///
536    /// # Errors
537    /// * An error occured during encoding.
538    fn encode_sequence(sequence: &ImageSequence<P>, dest: W) -> crate::Result<()> {
539        let mut encoder = Self::new(dest, sequence)?;
540        for frame in sequence.iter() {
541            encoder.add_frame(frame)?;
542        }
543        encoder.finish()
544    }
545}

547/// Low-level decoder interface around an image format.
548pub trait Decoder<P: Pixel, R: Read> {
549    /// The type of the iterator returned by `decode_sequence`.
550    type Sequence: FrameIterator<P>;
551
552    /// Decodes the given stream into an image.
553    ///
554    /// # Errors
555    /// * An error occured during decoding.
556    fn decode(&mut self, stream: R) -> crate::Result<Image<P>>;
557
558    /// Decodes the given stream into a frame iterator.
559    ///
560    /// # Errors
561    /// * An error occured during decoding.
562    fn decode_sequence(&mut self, stream: R) -> crate::Result<Self::Sequence>;
563}

565/// Represents the lazy decoding of frames from an encoded image sequence, such as an animated
566/// image.
567///
568/// # See Also
569/// * [`ImageSequence`]
570/// * [`Frame`]
571pub trait FrameIterator<P: Pixel>: Iterator<Item = crate::Result<Frame<P>>> {
572    /// Returns the number of frames in the sequence.
573    ///
574    /// This does not consume any frames as this data is usually known from the very beginning of
575    /// decoding.
576    fn len(&self) -> u32;
577
578    /// Returns if there are no frames in the sequence. In this case, the image is probably
579    /// invalid to be encoded again.
580    fn is_empty(&self) -> bool {
581        self.len() == 0
582    }
583
584    /// Returns the amount of times this sequence will loop over itself.
585    fn loop_count(&self) -> LoopCount;
586
587    /// Collects all frames in this iterator and turns it into a high level [`ImageSequence`].
588    /// If any frame fails, that error is returned.
589    ///
590    /// # Errors
591    /// * An error occured during decoding one of the frames.
592    fn into_sequence(self: Box<Self>) -> crate::Result<ImageSequence<P>> {
593        let loop_count = self.loop_count();
594        let frames = self.collect::<crate::Result<Vec<_>>>()?;
595
596        Ok(ImageSequence::from_frames(frames).with_loop_count(loop_count))
597    }
598}

600/// Represents a single static image wrapped in a frame iterator.
601#[allow(clippy::large_enum_variant)]
602pub struct SingleFrameIterator<P: Pixel>(Option<Image<P>>);
603
604impl<P: Pixel> SingleFrameIterator<P> {
605    /// Create a new single static image frame iterator.
606    #[must_use]
607    pub const fn new(image: Image<P>) -> Self {
608        Self(Some(image))
609    }
610}
611
612impl<P: Pixel> FrameIterator<P> for SingleFrameIterator<P> {
613    fn len(&self) -> u32 {
614        1
615    }
616    fn loop_count(&self) -> LoopCount {
617        LoopCount::Exactly(1)
618    }
619
620    fn into_sequence(mut self: Box<Self>) -> crate::Result<ImageSequence<P>> {
621        let image = self.0.take().unwrap();
622        let frame = Frame::from_image(image);
623
624        Ok(ImageSequence::new().with_frame(frame))
625    }
626}
627
628impl<P: Pixel> Iterator for SingleFrameIterator<P> {
629    type Item = crate::Result<Frame<P>>;
630
631    fn next(&mut self) -> Option<Self::Item> {
632        self.0.take().map(|image| Ok(Frame::from_image(image)))
633    }
634
635    fn size_hint(&self) -> (usize, Option<usize>) {
636        if self.0.is_some() {
637            (1, Some(1))
638        } else {
639            (0, Some(0))
640        }
641    }
