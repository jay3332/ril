1//! Includes the [`Fill`] and [`IntoFill`] traits and primitive fill types such as [`SolidFill`].
2//!
3//! You should almost never need use a fill type directly since methods that take fills usually
4//! allow anything that implements [`IntoFill`], which includes things such as [`Pixel`]s for solid
5//! fills, [`gradient`][crate::gradient] types, and even [`Image`]s for image fills.
6
7use crate::{Image, OverlayMode, Pixel};
8
9pub type BoundingBox<T> = (T, T, T, T);
10
11/// Any fill type that can be used to fill a shape, i.e. solid colors or gradients.
12///
13/// For solid colors, this trait is implemented for all [`Pixel`] types as [`SolidFill`].
14pub trait IntoFill: Clone {
15    /// The pixel type of the fill.
16    type Pixel: Pixel;
17    /// The fill type.
18    type Fill: Fill<Self::Pixel>;
19
20    /// Converts the fill into a fill type.
21    fn into_fill(self) -> Self::Fill;
22}
23
24/// Handles the actual filling of a shape. See [`IntoFill`] for more information.
25pub trait Fill<P: Pixel>: Clone {
26    /// Sets the bounding box of the fill in place. This is used internally.
27    fn set_bounding_box(&mut self, _bounding_box: BoundingBox<u32>) {}
28
29    /// Sets the overlay mode of the fill. This is used internally.
30    #[must_use = "this method consumes the fill and returns it back, it does not modify it in-place"]
31    fn with_bounding_box(mut self, bounding_box: BoundingBox<u32>) -> Self
32    where
33        Self: Sized,
34    {
35        self.set_bounding_box(bounding_box);
36        self
37    }
38
39    /// Gets the color of the fill at the given coordinates.
40    fn get_pixel(&self, x: u32, y: u32) -> P;
41
42    /// Plots the fill at the given coordinates on the given image.
43    fn plot(&self, image: &mut Image<P>, x: u32, y: u32, mode: OverlayMode) {
44        image.overlay_pixel_with_mode(x, y, self.get_pixel(x, y), mode);
45    }
46
47    /// Plots the fill at the given coordinates on the given image with a custom alpha value.
48    fn plot_with_alpha(&self, image: &mut Image<P>, x: u32, y: u32, mode: OverlayMode, alpha: u8) {
49        image.overlay_pixel_with_alpha(x, y, self.get_pixel(x, y), mode, alpha);
50    }
51}
52
53impl<P: Pixel> IntoFill for P {
54    type Pixel = P;
55    type Fill = SolidFill<Self::Pixel>;
56
57    fn into_fill(self) -> Self::Fill {
58        SolidFill(self)
59    }
60}
61
62impl<'a, P: Pixel> IntoFill for &'a Image<P> {
63    type Pixel = P;
64    type Fill = ImageFill<'a, Self::Pixel>;
65
66    fn into_fill(self) -> Self::Fill {
67        ImageFill(self)
68    }
69}

71/// Represents a solid color fill.
72#[derive(Copy, Clone, Debug)]
73pub struct SolidFill<P: Pixel>(P);
74
75impl<P: Pixel> SolidFill<P> {
76    /// Creates a new solid fill.
77    #[must_use]
78    pub const fn new(color: P) -> Self {
79        Self(color)
80    }
81
82    /// Returns the color (represented as a [`Pixel`]) of the fill.
83    #[must_use]
84    pub const fn color(&self) -> P {
85        self.0
86    }
87}
88
89impl<P: Pixel> Fill<P> for SolidFill<P> {
90    #[inline]
91    fn get_pixel(&self, _: u32, _: u32) -> P {
92        self.0
93    }
94}

96/// Represents a fill that pulls pixels from an image.
97///
98/// # Warning
99/// If pixels are unable to be pulled from this image (probably because it is too small),
100/// **the pixel at `(0, 0)` will be used as the fallback pixel**. It is important to know this since
101/// otherwise it can lead to confusing results.
102///
103/// As a rule of thumb, if you are unsure that the image in this fill matches or has higher
104/// dimensions than the object being drawn using this fill, resize the image to fit.
105#[derive(Copy, Clone)]
106pub struct ImageFill<'a, P: Pixel>(&'a Image<P>);
107
108impl<'a, P: Pixel> ImageFill<'a, P> {
109    /// Creates a new image fill from the given image.
110    #[must_use]
111    pub const fn new(image: &'a Image<P>) -> Self {
112        Self(image)
113    }
114
115    /// Returns a reference to the host image.
116    #[must_use]
117    pub const fn image(&self) -> &'a Image<P> {
118        self.0
119    }
120}
121
122impl<P: Pixel> Fill<P> for ImageFill<'_, P> {
123    fn get_pixel(&self, x: u32, y: u32) -> P {
124        self.0
125            .get_pixel(x, y)
126            .copied()
127            .unwrap_or_else(|| self.0.data[0])
128    }
129}
