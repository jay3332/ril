17impl From<png::ColorType> for ColorType {
18    #[allow(clippy::enum_glob_use)]
19    fn from(value: png::ColorType) -> Self {
20        use png::ColorType::*;
21
22        match value {
23            Grayscale => Self::Luma,
24            GrayscaleAlpha => Self::LumaA,
25            Rgb => Self::Rgb,
26            Rgba => Self::Rgba,
27            Indexed => Self::PaletteRgb,
28        }
29    }
30}

32#[allow(clippy::enum_glob_use)]
33const fn get_png_color_type(src: ColorType) -> png::ColorType {
34    use png::ColorType::*;
35
36    match src {
37        ColorType::Luma => Grayscale,
38        ColorType::LumaA => GrayscaleAlpha,
39        ColorType::Rgb => Rgb,
40        ColorType::Rgba => Rgba,
41        ColorType::PaletteRgb | ColorType::PaletteRgba => Indexed,
42        ColorType::Dynamic => unreachable!(),
43    }
44}

90/// A PNG encoder interface around [`png::Encoder`].
91///
92/// # Note
93/// You **must** anticipate the frame and loop counts of the sequence with
94/// [`crate::EncoderMetadata::with_sequence`] before calling [`PngEncoder::add_frame`].
95/// See [`Encoder#anticipating-frame-and-loop-counts`] for more information.
96pub struct PngEncoder<P: Pixel, W: Write> {
97    writer: png::Writer<W>,
98    dimensions: (u32, u32),
99    _marker: PhantomData<P>,
100}
101
102impl<P: Pixel, W: Write> Encoder<P, W> for PngEncoder<P, W> {
103    type Config = PngEncoderOptions;
104
105    fn new(
106        dest: W,
107        metadata: impl encode::HasEncoderMetadata<Self::Config, P>,
108    ) -> crate::Result<Self> {
109        let mut encoder = png::Encoder::new(dest, metadata.width(), metadata.height());
110        encoder.set_color(get_png_color_type(metadata.color_type()));
111        encoder.set_depth(png::BitDepth::from_u8(metadata.bit_depth()).unwrap());
112
113        match metadata.color_type() {
114            ColorType::PaletteRgb => {
115                let pal = metadata.palette().expect("no palette for paletted image?");
116                encoder.set_palette(pal.iter().flat_map(Pixel::as_bytes).collect::<Cow<_>>());
117            }
118            ColorType::PaletteRgba => {
119                let pal = metadata.palette().expect("no palette for paletted image?");
120                encoder.set_palette(
121                    pal.iter()
122                        .map(Pixel::as_rgb)
123                        .flat_map(|p| p.as_bytes())
124                        .collect::<Cow<_>>(),
125                );
126                encoder.set_trns(pal.iter().map(|p| p.as_rgba().a).collect::<Cow<_>>());
127            }
128            _ => (),
129        }
130
131        if let Some((len, loops)) = metadata.sequence() {
132            encoder.set_animated(len as _, loops.count_or_zero())?;
133        }
134
135        let dimensions = (metadata.width(), metadata.height());
136        let config = metadata.config();
137        encoder.set_adaptive_filter(config.adaptive_filter);
138        encoder.set_filter(config.filter);
139        encoder.set_compression(config.compression);
140
141        Ok(Self {
142            writer: encoder.write_header()?,
143            dimensions,
144            _marker: PhantomData,
145        })
146    }
147
148    fn add_frame(&mut self, frame: &impl encode::FrameLike<P>) -> crate::Result<()> {
149        if (frame.image().dimensions()) != self.dimensions {
150            self.writer
151                .set_frame_dimension(frame.image().width(), frame.image().height())?;
152        }
153        let data = frame
154            .image()
155            .data
156            .iter()
157            .flat_map(P::as_bytes)
158            .collect::<Vec<_>>();
159
160        if let Some(delay) = frame.delay() {
161            self.writer
162                .set_frame_delay(delay.as_millis() as u16, 1000)?;
163        }
164        if let Some(disposal) = frame.disposal() {
165            self.writer.set_dispose_op(match disposal {
166                DisposalMethod::None => png::DisposeOp::None,
167                DisposalMethod::Background => png::DisposeOp::Background,
168                DisposalMethod::Previous => png::DisposeOp::Previous,
169            })?;
170        }
171
172        self.writer.write_image_data(&data)?;
173        Ok(())
174    }
175
176    fn finish(self) -> crate::Result<()> {
177        self.writer.finish()?;
178        Ok(())
179    }
180}

182/// A PNG decoder interface around [`png::Decoder`].
183pub struct PngDecoder<P: Pixel, R: Read> {
184    _marker: PhantomData<(P, R)>,
185}
186
187impl<P: Pixel, R: Read> PngDecoder<P, R> {
188    /// Creates a new decoder with the default settings.
189    #[must_use]
190    pub const fn new() -> Self {
191        Self {
192            _marker: PhantomData,
193        }
194    }
195
196    fn prepare(stream: R) -> crate::Result<png::Reader<R>> {
197        let decoder = png::Decoder::new(stream);
198        decoder.read_info().map_err(Into::into)
199    }
200}

202#[allow(clippy::type_complexity)]
203fn read_data<P: Pixel>(
204    buffer: &[u8],
205    info: &png::Info,
206) -> crate::Result<(Vec<P>, Option<Box<[P::Color]>>)> {
207    let color_type: ColorType = info.color_type.into();
208    let bit_depth = info.bit_depth as u8;
209
210    let palette = info.palette.as_deref().map(|pal| {
211        let pal = pal
212            .chunks_exact(3)
213            .map(|p| (p[0], p[1], p[2]))
214            .collect::<Vec<_>>();
215
216        let pal = if let Some(trns) = info.trns.as_deref() {
217            trns.iter()
218                .zip(pal)
219                .map(|(a, (r, g, b))| PaletteRepr::Rgba(r, g, b, *a))
220                .collect::<Vec<_>>()
221        } else {
222            pal.into_iter()
223                .map(|(r, g, b)| PaletteRepr::Rgb(r, g, b))
224                .collect()
225        };
226
227        pal.into_iter()
228            .map(|p| match p {
229                PaletteRepr::Rgb(r, g, b) => Dynamic::Rgb(Rgb::new(r, g, b)),
230                PaletteRepr::Rgba(r, g, b, a) => Dynamic::Rgba(Rgba::new(r, g, b, a)),
231            })
232            .map(P::Color::from_dynamic)
233            .collect::<Vec<_>>()
234            .into_boxed_slice()
235    });
236
237    let chunks = buffer.chunks_exact(info.bytes_per_pixel());
238    let data = if P::COLOR_TYPE.is_paletted() {
239        let palette = palette.as_deref().expect("no palette for paletted image?");
240        chunks
241            // SAFETY: considered safe for unartificial types as the safety is upheld by the
242            // crate. Otherwise, safety must be upheld by the user.
243            .map(|idx| unsafe { assume_pixel_from_palette(palette, idx[0]) })
244            .collect::<crate::Result<Vec<_>>>()?
245    } else {
246        chunks
247            .map(|chunk| {
248                P::from_raw_parts_paletted(color_type, bit_depth, chunk, palette.as_deref())
249            })
250            .collect::<crate::Result<Vec<_>>>()?
251    };
252
253    Ok((data, palette))
254}
255
256enum PaletteRepr {
257    Rgb(u8, u8, u8),
258    Rgba(u8, u8, u8, u8),
259}

261impl<P: Pixel, R: Read> Decoder<P, R> for PngDecoder<P, R> {
262    type Sequence = ApngFrameIterator<P, R>;
263
264    fn decode(&mut self, stream: R) -> crate::Result<Image<P>> {
265        let mut reader = Self::prepare(stream)?;
266
267        // Here we are decoding a single image, so only capture the first frame:
268        let buffer = &mut vec![0; reader.output_buffer_size()];
269        reader.next_frame(buffer)?;
270
271        let info = reader.info();
272        let (data, palette) = read_data(buffer, info)?;
273
274        Ok(Image {
275            width: NonZeroU32::new(info.width).unwrap(),
276            height: NonZeroU32::new(info.height).unwrap(),
277            data,
278            format: ImageFormat::Png,
279            overlay: OverlayMode::default(),
280            palette,
281        })
282    }
283
284    fn decode_sequence(&mut self, stream: R) -> crate::Result<Self::Sequence> {
285        let reader = Self::prepare(stream)?;
286
287        Ok(ApngFrameIterator {
288            seq: 0,
289            reader,
290            _marker: PhantomData,
291        })
292    }
293}

301pub struct ApngFrameIterator<P: Pixel, R: Read> {
302    seq: u32,
303    reader: png::Reader<R>,
304    _marker: PhantomData<P>,
305}
306
307impl<P: Pixel, R: Read> ApngFrameIterator<P, R> {
308    fn info(&self) -> &png::Info<'_> {
309        self.reader.info()
310    }
311
312    #[allow(clippy::type_complexity)]
313    fn next_frame(&mut self) -> crate::Result<(Vec<P>, Option<Box<[P::Color]>>, png::OutputInfo)> {
314        let buffer = &mut vec![0; self.reader.output_buffer_size()];
315        let info = self.reader.next_frame(buffer)?;
316        let (data, palette) = read_data(buffer, self.info())?;
317
318        Ok((data, palette, info))
319    }
320}

322impl<P: Pixel, R: Read> FrameIterator<P> for ApngFrameIterator<P, R> {
323    fn len(&self) -> u32 {
324        self.info().animation_control.map_or(1, |a| a.num_frames)
325    }
326
327    fn loop_count(&self) -> LoopCount {
328        match self.info().animation_control.map(|a| a.num_plays) {
329            Some(0) | None => LoopCount::Infinite,
330            Some(n) => LoopCount::Exactly(n),
331        }
332    }
333}
334
335impl<P: Pixel, R: Read> Iterator for ApngFrameIterator<P, R> {
336    type Item = crate::Result<Frame<P>>;
337
338    fn next(&mut self) -> Option<Self::Item> {
339        if self.seq >= self.len() {
340            return None;
341        }
342
343        let (data, palette, output_info) = match self.next_frame() {
344            Ok(o) => o,
345            Err(e) => return Some(Err(e)),
346        };
347
348        let inner = Image {
349            width: NonZeroU32::new(output_info.width).unwrap(),
350            height: NonZeroU32::new(output_info.height).unwrap(),
351            data,
352            format: ImageFormat::Png,
353            overlay: OverlayMode::default(),
354            palette,
355        };
356
357        self.seq += 1;
358        let fc = self.info().frame_control();
359
360        Some(Ok(Frame::from_image(inner)
361            .with_delay(fc.map_or_else(Duration::default, |f| {
362                Duration::from_secs_f64(f64::from(f.delay_num) / f64::from(f.delay_den))
363            }))
364            .with_disposal(fc.map_or_else(
365                DisposalMethod::default,
366                |f| match f.dispose_op {
367                    png::DisposeOp::None => DisposalMethod::None,
368                    png::DisposeOp::Background => DisposalMethod::Background,
369                    png::DisposeOp::Previous => DisposalMethod::Previous,
370                },
371            ))))
372    }
373
374    fn size_hint(&self) -> (usize, Option<usize>) {
375        let remaining = (self.len() - self.seq) as usize;
376
377        (remaining, Some(remaining))
378    }
