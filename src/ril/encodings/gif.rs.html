13/// Options for encoding GIFs.
14#[derive(Copy, Clone, Debug, PartialEq, Eq)]
15pub struct GifEncoderOptions {
16    speed: u8,
17}
18
19impl Default for GifEncoderOptions {
20    fn default() -> Self {
21        Self::new()
22    }
23}

25impl GifEncoderOptions {
26    /// Creates a new encoder with default settings.
27    #[must_use]
28    pub const fn new() -> Self {
29        Self { speed: 10 }
30    }
31
32    /// Sets the speed of the encoder. Higher speeds come at the cost of lower image quality.
33    /// Must be between 1 and 30.
34    ///
35    /// # Panics
36    /// * Speed is not between 1 and 30.
37    #[must_use]
38    pub const fn with_speed(mut self, speed: u8) -> Self {
39        debug_assert!(speed > 0 && speed <= 30, "speed must be between 1 and 30");
40
41        self.speed = speed;
42        self
43    }
44}

46/// A GIF encoder interface over [`gif::Encoder`].
47pub struct GifEncoder<P: Pixel, W: Write> {
48    options: GifEncoderOptions,
49    encoder: gif::Encoder<W>,
50    _marker: PhantomData<P>,
51}

53impl<P: Pixel, W: Write> GifEncoder<P, W> {
54    #[allow(clippy::cast_lossless)]
55    fn build_frame<'a>(&self, image: &Image<P>) -> crate::Result<gif::Frame<'a>> {
56        macro_rules! data {
57            ($t:ty) => {{
58                image
59                    .data
60                    .iter()
61                    .flat_map(|p| <$t>::from(Dynamic::from_pixel(*p).unwrap()).as_bytes())
62                    .collect::<Vec<_>>()
63            }};
64            () => {{
65                image
66                    .data
67                    .iter()
68                    .flat_map(|p| p.as_bytes())
69                    .collect::<Vec<_>>()
70            }};
71        }
72
73        macro_rules! rgb {
74            ($data:expr) => {{
75                let pixels = $data;
76
77                gif::Frame::from_rgb_speed(
78                    image.width() as u16,
79                    image.height() as u16,
80                    &pixels,
81                    self.options.speed as i32,
82                )
83            }};
84        }
85
86        macro_rules! rgba {
87            ($data:expr) => {{
88                let mut pixels = $data;
89
90                gif::Frame::from_rgba_speed(
91                    image.width() as u16,
92                    image.height() as u16,
93                    &mut pixels,
94                    self.options.speed as i32,
95                )
96            }};
97        }
98
99        Ok(match (image.data[0].color_type(), P::BIT_DEPTH) {
100            (ColorType::Rgb, 8) => rgb!(data!()),
101            (ColorType::Rgba, 8) => rgba!(data!()),
102            (ColorType::Luma, 1 | 8) => rgb!(data!(crate::Rgb)),
103            (ColorType::LumaA, 1 | 8) => rgba!(data!(Rgba)),
104            (ColorType::PaletteRgb, 8) => gif::Frame::from_palette_pixels(
105                image.width() as u16,
106                image.height() as u16,
107                image
108                    .data
109                    .iter()
110                    .flat_map(|p| p.as_rgb().as_bytes())
111                    .collect::<Vec<_>>()
112                    .as_slice(),
113                image
114                    .palette()
115                    .expect("paletted image without palette?")
116                    .iter()
117                    .flat_map(|p| p.as_rgb().as_bytes())
118                    .collect::<Vec<_>>()
119                    .as_slice(),
120                None,
121            ),
122            (ColorType::PaletteRgba, 8) => {
123                let pixels = image.palette().expect("paletted image without palette?");
124                // TODO: flatten all transparent pixels to the same color
125                let transparent_index = pixels
126                    .iter()
127                    .position(|p| p.as_rgba().a == 0)
128                    .map(|i| i as u8);
129
130                gif::Frame::from_palette_pixels(
131                    image.width() as u16,
132                    image.height() as u16,
133                    image
134                        .data
135                        .iter()
136                        .flat_map(|p| p.as_rgba().as_bytes())
137                        .collect::<Vec<_>>()
138                        .as_slice(),
139                    pixels
140                        .iter()
141                        .flat_map(|p| p.as_rgb().as_bytes())
142                        .collect::<Vec<_>>()
143                        .as_slice(),
144                    transparent_index,
145                )
146            }
147            _ => return Err(Error::UnsupportedColorType),
148        })
149    }
150}

152impl<P: Pixel, W: Write> Encoder<P, W> for GifEncoder<P, W> {
153    type Config = GifEncoderOptions;
154
155    fn new(
156        dest: W,
157        metadata: impl encode::HasEncoderMetadata<Self::Config, P>,
158    ) -> crate::Result<Self> {
159        let mut encoder =
160            gif::Encoder::new(dest, metadata.width() as u16, metadata.height() as u16, &[])?;
161
162        if let Some((_, loop_count)) = metadata.sequence() {
163            encoder.set_repeat(match loop_count {
164                LoopCount::Exactly(n) => gif::Repeat::Finite(n as u16),
165                LoopCount::Infinite => gif::Repeat::Infinite,
166            })?;
167        }
168
169        Ok(Self {
170            options: metadata.config(),
171            encoder,
172            _marker: PhantomData,
173        })
174    }
175
176    #[allow(clippy::cast_precision_loss)]
177    fn add_frame(&mut self, frame: &impl encode::FrameLike<P>) -> crate::Result<()> {
178        let mut out = self.build_frame(frame.image())?;
179
180        if let Some(delay) = frame.delay() {
181            out.delay = (delay.as_millis() as f64 / 10.).round() as u16;
182        }
183        if let Some(disposal) = frame.disposal() {
184            out.dispose = match disposal {
185                DisposalMethod::None => gif::DisposalMethod::Keep,
186                DisposalMethod::Background => gif::DisposalMethod::Background,
187                DisposalMethod::Previous => gif::DisposalMethod::Previous,
188            };
189        }
190
191        self.encoder.write_frame(&out)?;
192        Ok(())
193    }
194
195    // no-op
196    fn finish(self) -> crate::Result<()> {
197        Ok(())
198    }
199}

222fn read_frame<P: Pixel, R: Read>(
223    decoder: &mut gif::Decoder<R>,
224) -> Option<crate::Result<(&gif::Frame<'_>, Image<P>)>> {
225    let global_palette = decoder.global_palette().map(ToOwned::to_owned);
226    let frame = match decoder.read_next_frame() {
227        Ok(Some(frame)) => frame,
228        Ok(None) => return None,
229        Err(e) => return Some(Err(e.into())),
230    };
231    let raw_palette = match frame.palette {
232        Some(ref palette) => palette,
233        None => match global_palette {
234            Some(ref palette) => palette,
235            None => return Some(Err(Error::DecodingError("missing palette".to_string()))),
236        },
237    };
238    let transparent_index = frame.transparent.map(|i| i as usize);
239
240    let palette = raw_palette
241        .chunks_exact(3)
242        .enumerate()
243        .map(|(i, p)| {
244            P::Color::from_dynamic(Dynamic::Rgba(Rgba {
245                r: p[0],
246                g: p[1],
247                b: p[2],
248                a: if Some(i) == transparent_index { 0 } else { 255 },
249            }))
250        })
251        .collect::<Vec<_>>();
252
253    let data = match frame
254        .buffer
255        .iter()
256        .map(|&i| unsafe { assume_pixel_from_palette(&palette, i) })
257        .collect::<crate::Result<Vec<_>>>()
258    {
259        Ok(data) => data,
260        Err(e) => return Some(Err(e)),
261    };
262
263    #[allow(clippy::cast_lossless)]
264    let width = frame.width as u32;
265    #[allow(clippy::cast_lossless)]
266    let height = frame.height as u32;
267
268    Some(Ok((
269        frame,
270        Image {
271            width: NonZeroU32::new(width).unwrap(),
272            height: NonZeroU32::new(height).unwrap(),
273            data,
274            format: ImageFormat::Gif,
275            overlay: OverlayMode::default(),
276            palette: P::COLOR_TYPE
277                .is_paletted()
278                .then(|| palette.into_boxed_slice()),
279        },
280    )))
281}

283impl<P: Pixel, R: Read> Decoder<P, R> for GifDecoder<P, R> {
284    type Sequence = GifFrameIterator<P, R>;
285
286    #[allow(clippy::cast_lossless)]
287    fn decode(&mut self, stream: R) -> crate::Result<Image<P>> {
288        let mut decoder = gif::DecodeOptions::new();
289        decoder.set_color_output(gif::ColorOutput::Indexed);
290        let mut decoder = decoder.read_info(stream)?;
291
292        Ok(read_frame(&mut decoder)
293            .unwrap_or(Err(Error::EmptyImageError))?
294            .1)
295    }
296
297    fn decode_sequence(&mut self, stream: R) -> crate::Result<Self::Sequence> {
298        let mut decoder = gif::DecodeOptions::new();
299        decoder.set_color_output(gif::ColorOutput::Indexed);
300
301        Ok(GifFrameIterator {
302            decoder: decoder.read_info(stream)?,
303            _marker: PhantomData,
304        })
305    }
306}

325impl<P: Pixel, R: Read> Iterator for GifFrameIterator<P, R> {
326    type Item = crate::Result<Frame<P>>;
327
328    #[allow(clippy::cast_lossless)]
329    fn next(&mut self) -> Option<Self::Item> {
330        let (frame, image) = match read_frame(&mut self.decoder)? {
331            Ok(image) => image,
332            Err(e) => return Some(Err(e)),
333        };
334
335        Some(Ok(Frame::from_image(image)
336            .with_delay(Duration::from_millis(frame.delay as u64 * 10))
337            .with_disposal(match frame.dispose {
338                gif::DisposalMethod::Keep | gif::DisposalMethod::Any => DisposalMethod::None,
339                gif::DisposalMethod::Background => DisposalMethod::Background,
340                gif::DisposalMethod::Previous => DisposalMethod::Previous,
341            })))
342    }
