5    images::{Image as ImageOut, ImageRef},

7    Resizer,
8};

10/// A filtering algorithm that is used to resize an image.
11#[derive(Copy, Clone, Debug, PartialEq, Eq)]
12pub enum FilterType {
13    /// A simple nearest neighbor algorithm. Although the fastest, this gives the lowest quality
14    /// resizings.
15    ///
16    /// When upscaling this is good if you want a "pixelated" effect with no aliasing.
17    Nearest,
18    /// A box filter algorithm. Equivalent to the [`Nearest`] filter if you are upscaling.
19    Box,
20    /// A bilinear filter. Calculates output pixel value using linear interpolation on all pixels.
21    Bilinear,
22    /// While having similar performance as the [`Bilinear`] filter, this produces a sharper and
23    /// usually considered better quality image than the [`Bilinear`] filter, but **only** when
24    /// downscaling. This may give worse results than bilinear when upscaling.
25    Hamming,
26    /// A Catmull-Rom bicubic filter, which is the most common bicubic filtering algorithm. Just
27    /// like all cubic filters, it uses cubic interpolation on all pixels to calculate output
28    /// pixels.
29    Bicubic,
30    /// A Mitchell-Netravali bicubic filter. Just like all cubic filters, it uses cubic
31    /// interpolation on all pixels to calculate output pixels.
32    Mitchell,
33    /// A Lanczos filter with a window of 3. Calculates output pixel value using a high-quality
34    /// Lanczos filter on all pixels. This can give antialiasing effects.
35    Lanczos3,
36    /// If upscaling, repeats the image in a tiling fashion to fill the desired size.
37    /// If downscaling, this just crops the image.
38    Tile,
39}
40
41impl Default for FilterType {
42    fn default() -> Self {
43        Self::Nearest
44    }
45}
46
47impl From<FilterType> for ResizeAlg {
48    fn from(f: FilterType) -> Self {
49        type F = ResizeFilterType;
50
51        Self::Convolution(match f {
52            FilterType::Nearest => return Self::Nearest,
53            FilterType::Box => F::Box,
54            FilterType::Bilinear => F::Bilinear,
55            FilterType::Hamming => F::Hamming,
56            FilterType::Bicubic => F::CatmullRom,
57            FilterType::Mitchell => F::Mitchell,
58            FilterType::Lanczos3 => F::Lanczos3,
59            FilterType::Tile => unimplemented!("tile filter is implemented separately"),
60        })
61    }
62}
63
64impl FilterType {
65    /// Performs a resize operation on the given data.
66    ///
67    /// # Panics
68    /// * The given data is empty.
69    /// * Unsupported bit depth.
70    pub fn resize<P: Pixel>(
71        &self,
72        data: &[P],
73        src_width: u32,
74        src_height: u32,
75        dst_width: u32,
76        dst_height: u32,
77    ) -> Vec<P> {
78        let color_type = data[0].color_type();
79        let pixel_type = match P::BIT_DEPTH {
80            1 | 2 | 4 | 8 => match color_type {
81                ColorType::Luma | ColorType::PaletteRgb | ColorType::PaletteRgba => {
82                    ResizePixelType::U8
83                }
84                ColorType::LumaA => ResizePixelType::U8x2,
85                ColorType::Rgb => ResizePixelType::U8x3,
86                ColorType::Rgba => ResizePixelType::U8x4,
87                ColorType::Dynamic => unreachable!(),
88            },
89            16 => match color_type {
90                ColorType::Luma | ColorType::PaletteRgb | ColorType::PaletteRgba => {
91                    ResizePixelType::U16
92                }
93                ColorType::LumaA => ResizePixelType::U16x2,
94                ColorType::Rgb => ResizePixelType::U16x3,
95                ColorType::Rgba => ResizePixelType::U16x4,
96                ColorType::Dynamic => unreachable!(),
97            },
98            _ => panic!("Unsupported bit depth"),
99        };
100
101        let buffer = data.iter().flat_map(P::as_bytes).collect::<Vec<_>>();
102        // We are able to unwrap here since we validated the buffer throughout the creation of the image.
103        let src =
104            ImageRef::new(src_width, src_height, &buffer, pixel_type).expect("Invalid buffer size");
105        let mut dest = ImageOut::new(dst_width, dst_height, pixel_type);
106
107        let mut resizer = Resizer::new();
108        let options = ResizeOptions::new().resize_alg(ResizeAlg::from(*self));
109
110        // The pixel type is the same, we can unwrap here
111        resizer.resize(&src, &mut dest, Some(&options)).unwrap();
112
113        let bpp = color_type.channels() * ((P::BIT_DEPTH as usize + 7) >> 3);
114        dest.into_vec()
115            .chunks_exact(bpp)
116            .map(P::from_bytes)
117            .collect()
118    }
119}
120
121fn resize_tiled<P: Pixel>(data: &[P], src_width: u32, dst_width: u32, dst_height: u32) -> Vec<P> {
122    let chunks = data.chunks_exact(src_width as _);
123
124    chunks
125        .flat_map(|chunk| chunk.iter().cycle().take(dst_width as _))
126        .cycle()
127        .take((dst_width * dst_height) as _)
128        .copied()
129        .collect()
130}
131
132/// Performs a resize operation on the given data.
133///
134/// # Panics
135/// * The given data is empty.
136/// * Unsupported bit depth.
137pub fn resize<P: Pixel>(
138    data: &[P],
139    src_width: u32,
140    src_height: u32,
141    dst_width: u32,
142    dst_height: u32,
143    filter: FilterType,
144) -> Vec<P> {
145    match filter {
146        FilterType::Tile => resize_tiled(data, src_width, dst_width, dst_height),
147        _ => filter.resize(data, src_width, src_height, dst_width, dst_height),
148    }
149}
